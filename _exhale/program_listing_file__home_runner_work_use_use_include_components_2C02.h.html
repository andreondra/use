    struct{

        uint8_t primaryOAM[256];

        // Attributes.

        // X positions.
        uint8_t x[8];

        uint8_t allowShift[8];

        uint8_t secondarySpriteId;

        uint8_t feedY;
        uint16_t feedTileAddress;
        uint8_t feedIndex;

        void renderInit(){

            memset(secondaryOAM, 0xFF, 32);

            secondarySpriteId = 0;
            feedIndex = 0;
        }

        void shiftClear(){

            memset(x, 0, 8);
            memset(shiftHi, 0, 8);
            memset(shiftLo, 0, 8);
            memset(attrLatch, 0, 8);
            memset(allowShift, 0, 8);
        }

        void shift(){

            for(uint8_t i = 0; i < 8; i++)
                if(allowShift[i]){
                    if((attrLatch[i] & 0x40) > 0){

                        shiftHi[i] >>= 1;
                        shiftLo[i] >>= 1;
                    } else {

                        shiftHi[i] <<= 1;
                        shiftLo[i] <<= 1;
                    }
                }
        }

        void clear(){

            memset(primaryOAM, 0, 256);
            shiftClear();
            renderInit();
        }
